class Cliente <<Client>>{ 
    -director : DirectorDeSandwichs

    +constructorDeClasico()
    +constructorDeVegetariano()
    +constructorDeVegano()
    +constructorDeSinTacc()
    
    +crearDirector(constructor : Constructor) : DirectorDeSandwichs
}

note left of Cliente 
    -en la realidad seria la interfaz de usuario

class DirectorDeSandwichs <<Director>>{
    -constructor : Constructor

    +<<create>> DirectorDeSandwichs(constructor : Constructor)
    +construirSandwich(constructor : Constructor) : Sandwich
}

abstract class ConstructorDeSandwich <<Constructor>>{
    +<<abstract>> construirPan() : Pan
    +<<abstract>> construirAderezo() : Aderezo
    +<<abstract>> construirPrincipal() : Principal
    +<<abstract>> construirAdicional() : Adicional
}

class ConstructorDeClasico <<ConcreteBuilder>> extends ConstructorDeSandwich{
    +construirPan() : PanBrioche
    +construirAderezo() : AderezoMayonesa
    +construirPrincipal() : PrincipalCarneTernera
    +construirAdicional() : AdicionalTomate

    +obtenerSandwich() : SandwichClasico
}

class ConstructorDeVegetariano <<ConcreteBuilder>> extends ConstructorDeSandwich{
    +construirPan() : PanSemiilas
    +construirAderezo() : AderezoNULL
    +construirPrincipal() : PrincipalProvoleta
    +construirAdicional() : AdicionalBerenjenasEscabeche

    +obtenerSandwich() : SandwichVegetariano
}

class ConstructorDeVegano <<ConcreteBuilder>> extends ConstructorDeSandwich{
    +construirPan() : PanIntegral
    +construirAderezo() : AderezoSalsaCriolla
    +construirPrincipal() : PrincipalMilanesaGirgolas
    +construirAdicional() : AdicionalNULL

    +obtenerSandwich() : SandwichVegano
}

class ConstructorDeSinTacc <<ConcreteBuilder>> extends ConstructorDeSandwich{
    +construirPan() : PanChipa
    +construirAderezo() : AderezoSalsaTartara
    +construirPrincipal() : PrincipalCarnePollo
    +construirAdicional() : AdicionalVerdurasGrilladas

    +obtenerSandwich() : SandwichSinTacc
}
