@startuml

abstract class Empleado{
    -nombre : String
    -apellido : String

    +sueldo() : Real
    +abstract basico() : Real
    +abstract adicional() : Real
    +descuento() : Real

}

class EmpleadoTemporario{
  -canthoras : Integer
  -cantHijos : Integer
    +basico() : Real
    +adicional() : Real

}

class EmpleadoPlanta{

    +basico() : Real
    +adicional() : Real

}

class EmpleadoPasante{
  -examenRendido : Integer
    +basico() : Real
    +adicional() : Real

}




EmpleadoPasante --|> Empleado
EmpleadoTemporario --|> Empleado
EmpleadoPlanta --|> Empleado


@enduml
